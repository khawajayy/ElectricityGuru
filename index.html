First-time setup

  1. Create a free account at supabase.com → New project (any name, pick a region near you, set a database password you keep).
  2. In the project: open SQL Editor, paste the script below, click Run. This creates your tables and locks them to your account.
  3. Open Project Settings → API. Copy the Project URL and the anon public key into the boxes on the left, then Save connection.
  4. To sign in instantly with no confirmation email: open Authentication → Sign In / Providers → Email and turn Confirm email OFF. (Optional but recommended.)
  5. Back here, enter an email + password and click Create account. Use the same details to sign in on your phone. Done, your data syncs everywhere.

create table readings (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null default auth.uid() references auth.users,
  v double precision not null,
  t bigint not null,
  created_at timestamptz default now()
);
create table appliances (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null default auth.uid() references auth.users,
  name text not null,
  watts double precision not null,
  qty integer not null default 1,
  hours double precision not null
);
alter table readings enable row level security;
alter table appliances enable row level security;
create policy "own_readings" on readings
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);
create policy "own_appliances" on appliances
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);
